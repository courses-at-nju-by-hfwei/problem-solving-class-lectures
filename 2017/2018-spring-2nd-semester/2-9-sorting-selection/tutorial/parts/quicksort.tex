\begin{frame}{}
  \begin{exampleblock}{Best-Case Complexity of \textsc{Quicksort} ($7.4-2$)}
    Show that \textsc{Quicksort}'s \red{\it best-case} running time is $\Omega(n \log n)$.
  \end{exampleblock}

  \pause
  \[
    T(n) = 2T(\frac{n}{2}) + \Theta(n)
  \]

  \pause
  \[
    T(n) = \underbrace{n}_{\textsc{\blue{Partition}}} \underbrace{\log n}_{\text{\blue{Height}}} \quad \text{(Recursion Tree)}
  \]

  \pause
  \vspace{0.30cm}
  \[
    \purple{\boxed{T(n) = \min_{0 \le q \le n-1} \Big( T(q) + T(n-q-1) \Big) + \Theta(n)}}
  \]

  \pause
  \[
    T(n) = \Omega(n \log n)
  \]

  \pause
  \centerline{\teal{\large By substitution.} {\footnotesize (Section 7.4.1, CLRS)}}
\end{frame}
