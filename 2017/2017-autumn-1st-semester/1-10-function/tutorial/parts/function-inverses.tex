\begin{frame}{}
  \centerline{\LARGE Inverse}

  \begin{definition}[Inverse]
    Let $f: A \to B$ be a \blue{bijective} function.

    The \red{inverse} of $f$ is the function $f^{-1}: B \to A$ defined by

    \[
      f^{-1}(y) = x \iff f(x) = y.
    \]
  \end{definition}
\end{frame}
%%%%%%%%%%%%%%%

%%%%%%%%%%%%%%%
\begin{frame}{}
  \begin{exampleblock}{(UD $15.11$)}
    \[
      f: A \to B
    \]

    \[
      g_1, g_2: B \to A
    \]

    \begin{enumerate}[(i)]
      \item 
	\[
	  f \circ g_1 = f \circ g_2, \quad f \text{ is bijective} \implies g_1 = g_2
	\]
      \item 
	\[
	  g_1 \circ f = g_2 \circ f, \quad f \text{ is bijective} \implies g_1 = g_2
	\]
    \end{enumerate}
  \end{exampleblock}
\end{frame}
