\begin{algorithm}[H]
  % \caption{Bellam-Ford Algorithm as a DP Algorithm.}
  % \label{alg:bellman-ford-dp}
  \begin{algorithmic}[1]
    \Procedure{Bellman-Ford-DP}{$G, w, s$}
      \State $d[0,s] \gets 0$
      \For{$(v \neq s) \in V$}
	\State $d[0,v] \gets \infty$
      \EndFor

      \hStatex
      \For{$i \gets 1 \;\text{\bf to}\; |V| - 1$}
	\For{$v \in V$}
	  \State $d[i,v] = \infty$
	  \For{$(u,v) \in E$}
	    \If{$d[i, v] > d[i-1, u] + w(u, v)$} \Comment{\onslide<2->{\textcolor{red}{Simplify?}}}
	      \State $d[i, v] = d[i-1, u] + w(u, v)$
	    \EndIf
	  \EndFor
	\EndFor
      \EndFor
    \EndProcedure
  \end{algorithmic}
\end{algorithm}
