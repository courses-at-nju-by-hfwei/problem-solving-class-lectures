\begin{frame}{}
  \centerline{\teal{\large Simulating a queue $Q$ using two stacks $S_1, S_2$ {\small (Problem $\mathbb{E}3$)}}}

  \begin{algorithm}[H]
    \begin{algorithmic}[]
      \Procedure{Enq}{$x$}
	\State \textsl{Push}($S_1, x$)
      \EndProcedure

      \Statex
      \Procedure{Deq}{\null}
        \If{$S_2 = \emptyset$}
          \While{$S_1 \neq \emptyset$}
            \State \textsl{Push}($S_2$, \textsl{Pop}($S_1$))
          \EndWhile
        \EndIf
        \textsl{Pop}($S_2$)
      \EndProcedure
    \end{algorithmic}
  \end{algorithm}
\end{frame}
