      \E[X] = \sum_{x} x \Pr(X = x)

  \end{definition}

  \begin{theorem}[Computing Expectation]
    Let $X$ be a discrete random variable that takes on \red{only nonnegative integer values}.

    \[
      \E[X] = \sum_{i=1}^{\infty} i \Pr(X = i) = \sum_{i=1}^{\infty} \Pr(X \le i)
    \]
  \end{theorem}
