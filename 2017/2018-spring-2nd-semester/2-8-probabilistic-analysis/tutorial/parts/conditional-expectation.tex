    \[
      \E[X \mid E] = \sum_{x} x \Pr(X = x \mid E)
    \]
  \end{definition}

  \vspace{0.50cm}
  \begin{definition}[Conditional Expectation \red{on a Random Variable}]
    \[
      \E[X \mid Y = y] = \sum_{x} x \Pr(X = x \mid Y = y)
    \]
  \end{definition}
\end{frame}
