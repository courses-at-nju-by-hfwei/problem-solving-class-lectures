\begin{frame}{}
  \begin{definition}[Stackable Permutations]
    \begin{description}[\texttt{push(X, S)}:]
      \item[\texttt{read(X)}:] $\texttt{>> } X$
      \item[\texttt{print(X)}:] $\texttt{<< } X$
      \item[\texttt{push(X, S)}:] $S \Leftarrow X$
      \item[\texttt{pop(X, S)}:] $X \Leftarrow S$
      %	\pause
      % \item[\texttt{is-empty(S)}:] $S = \emptyset?$
    \end{description}
  \end{definition}

  \vspace{0.30cm}
  \pause

\end{frame}
