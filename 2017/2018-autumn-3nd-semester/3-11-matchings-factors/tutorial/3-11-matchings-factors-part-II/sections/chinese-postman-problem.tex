\begin{frame}{}
  \begin{definition}[Chinese Postman Problem]
    \begin{center}
      Given an undirected weighted graph $G$ with \teal{$w(e) > 0$}, \\[5pt]
      to find the shortest \purple{tour} such that \red{each edge is traversed at least once}.
    \end{center}
  \end{definition}

  \pause
  \vspace{0.30cm}

\begin{frame}{}
  \begin{center}
    \teal{$P$: A postman tour of $G$} \\[5pt] \pause
    $P$ contains every edge $e$ at least once. \\[5pt] \pause
    Let \purple{$1 + x_{e}$} be the number of times edge $e$ is in $P$. \pause
    \[
      \blue{\text{Construct } G' = G + e \cdot x_{e}}
    \]
    \pause
    \red{\fbox{$P$ is an Eulerian tour of $G'$.}}
  \end{center}

  \pause
  \vspace{0.20cm}
  \begin{definition}[Chinese Postman Problem]
    Given an undirected weighted graph with \teal{$w(e) > 0$}, \\[5pt]
    to find $x_e \ge 0$ for each edge $e$ of $G$, such that \\
    $\sum_{e} w(e) x_e$ is minimized 
    
  \end{definition}
\end{frame}
