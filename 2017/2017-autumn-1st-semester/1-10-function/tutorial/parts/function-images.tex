\begin{frame}{}
  \centerline{\LARGE Images and Inverse Images}

  \[
    f: X \to Y, \quad A \subseteq X, \quad B \subseteq Y
  \]

  \[
    f(A) = \set{f(a): a \in A}
  \]

  \[
    f^{-1}(B) = \set{x \in X: f(x) \in B}
  \]
\end{frame}
