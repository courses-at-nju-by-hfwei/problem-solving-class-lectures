\begin{algorithm}[H]
  % \caption{Construct Optimal BST.}
  % \label{alg:construct-optimal-bst}
  \begin{algorithmic}[1]
    \Procedure{Construct-Optimal-BST}{$root$, $i$, $j$}
      \If{\red{$j = i-1$}}
	\State \Return node with \blue{$d_{i-1}$}
      \EndIf

      \hStatex
      \State $\purple{rn} \gets \text{ node with key } k_{root[i,j]}$
      \State $rn.\textsl{left} \gets \Call{Construct-Optimal-BST}{root, i, root[i,j]-1}$
      \State $rn.\textsl{right} \gets \Call{Construct-Optimal-BST}{root, root[i,j]+1, j}$

      \hStatex
      \State \Return $rn$
    \EndProcedure
  \end{algorithmic}
\end{algorithm}
