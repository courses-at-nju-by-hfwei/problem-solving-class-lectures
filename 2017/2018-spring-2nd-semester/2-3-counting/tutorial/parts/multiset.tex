\begin{frame}{}
  \begin{exampleblock}{Multisets ($\text{CS}: 1.5-4$)}
    Use multisets to determine the number of ways to pass out $k$ \red{\bf identical} apples to $n$ children. 
    Assume that a child may get more than one apple.
  \end{exampleblock}

  \vspace{0.30cm}
  \begin{columns}
    \column{0.30\textwidth}
      \fignocaption{width = 0.95\textwidth}{figs/apple-children}
    \column{0.70\textwidth}
      \pause
      \[
	x_i: \text{the $i$-th child gets $x_i$ apples}
      \]
      \[
	x_1 + x_2 + \cdots + x_n = k, \qquad x_i \ge 0
      \]
      \pause
      \[
	\red{y_i \triangleq x_i + 1}
      \]
      \pause
      \[
	y_1 + y_2 + \cdots + y_n = n + k, \qquad \red{y_i \le 1}
      \]
      \pause 
      \[
	\binom{n+k-1}{n-1} = \binom{n+k-1}{k}
      \]
  \end{columns}
\end{frame}
