\begin{algorithm}[H]
  \begin{algorithmic}[1]
    \Procedure{Stackable}{$out$}
      \ForAll{$a_j \in out$}
	\While{$\blue{\textsf{top}(S) \neq a_j}$}
	  \State \purple{$\textsf{Push}(in, S)$}
	\EndWhile
	\State \purple{$\textsf{Pop}(out, S)$}
      \EndFor
    \EndProcedure
  \end{algorithmic}
\end{algorithm}
