\begin{frame}
  \begin{exampleblock}{UD Problem 7.1 (d)}
    Let $A, B \subseteq X$.

    \[
      A \subseteq B \iff (X \setminus B) \subseteq (X \setminus A)
    \]
  \end{exampleblock}

  \pause
  \[
    \red{Q: A, B \subseteq X?}
  \]

  \pause
  \begin{columns}
    \column{0.50\textwidth}
      \[
	\blue{1.\; A \subseteq B \implies (X \setminus B) \subseteq (X \setminus A)}
      \]

      \uncover<4->{
	\begin{center}
	  \teal{By Contradiction.}
	\end{center}
      }
    \column{0.50\textwidth}
      \[
	\blue{2.\; (X \setminus B) \subseteq (X \setminus A) \implies A \subseteq B}
      \]

      \uncover<5->{
	\begin{center}
	  \teal{By Contradiction.}
	\end{center}
      }
  \end{columns}

  \uncover<6->{
    \vspace{0.80cm}
    \[
      \red{(2)\; \text{ needs } A \subseteq X}
    \]
  }
\end{frame}

\begin{frame}{}
  \begin{exampleblock}{UD Problem 7.14}
    \blue{Let $A, B \subseteq X$.}
    Prove that the union of two sets can be rewritten as the union of two \red{disjoint} sets.
    \begin{enumerate}[(a)]
      \item Prove that $(A \setminus B) \cap B = \emptyset$
      \item Prove that $A \cup B = (A \setminus B) \cup B$
    \end{enumerate}
  \end{exampleblock}

  \vspace{0.30cm}
  \begin{columns}
    \column{0.35\textwidth}
      \uncover<3->{
	\begin{center}
	  \blue{By Contradiction.}
	\end{center}
      }
    \column{0.35\textwidth}
      \pause

	\onslide<4->{\blue{(A \setminus B) \cup B = \ldots}} \\
	\onslide<5->{(A \cap \red{\bar{B}^{(X)}}) \cup B}

  \end{columns}
