    \begin{enumerate}[(a)]
      \item $\lnot(\lnot P)$
      \item $\lnot(P \lor Q)$
      \item $\lnot(P \land Q)$
      \item $P \to Q$
    \end{enumerate}
  \end{exampleblock}

  \vspace{0.30cm}
  \begin{columns}
    \column{0.40\textwidth}

	\item $P$
      \end{enumerate}
    \pause
    \column{0.40\textwidth}
      \begin{itemize}
	\item De Morgan's Law
	\item Implication
	  \[
	    (P \to Q) \leftrightarrow (\lnot P \lor Q)
	  \]
      \end{itemize}
