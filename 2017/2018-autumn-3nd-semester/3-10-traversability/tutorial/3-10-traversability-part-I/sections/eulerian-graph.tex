\begin{frame}{}
  \begin{theorem}[Leonhard Euler $1735$]
    A connected graph $G$ is \red{Eulerian} if and only if the degree of each vertex of $G$ is even.
  \end{theorem}

  \pause
  \begin{proof}[``$\Longleftarrow$'' (Carl Hierholzer $1873$); \pause\teal{\small (Proofs on PPT from Tao \& on CZ)}]
    \pause
    \begin{center}
      By strong mathematical \red{induction on the number $m$ of edges} of $G$.
    \end{center}
    \pause
    \[
      \text{Inductive Step}: m = k + 1
    \]

    \pause
    \begin{center}
      Let $T = u \sim v$ be any trail in $G$ of the maximum length.
    \end{center}
    \pause
    \[
      v = u \pause \implies T = u \sim u
    \]
    \pause
    \[
      \purple{H = G - E(T) = \bigcup H_i}
    \]
  \end{proof}
\end{frame}

      \]

      \end{center}
  \end{columns}
