\begin{frame}{Generalized Hamming codes}
  \begin{definition}[Generalized Hamming codes]
	\[
	  H_r \triangleq H_{r \times (2^r-1)}: \text{all nonzero binary vectors of length } r
	\]

	\[
	  C_{Ham}^{r} = (n = 2^r-1, k = 2^r-1-r)
	\]
  \end{definition}
