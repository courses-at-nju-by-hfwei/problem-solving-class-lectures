\begin{frame}{}
  \begin{definition}[Injective (one-to-one; 1-1)]
    \[
      \forall a_1, a_2 \in A: f(a_1) = f(a_2) \implies a_1 = a_2
    \]

    \[
      \forall a_1, a_2 \in A: a_1 \neq a_2 \implies f(a_1) \neq f(a_2)
    \]
  \end{definition}

  \pause
  \[
    \exists a_1, a_2 \in A: a_1 \neq a_2 \land f(a_1) = f(a_2)
  \]
\end{frame}
%%%%%%%%%%%%%%%

%%%%%%%%%%%%%%%
\begin{frame}{}
  \begin{definition}[Surjective (onto)]
    \[
      \forall b \in B: \exists a \in A: f(a) = b
    \]

    \[
      \ran{f} = B
    \]
  \end{definition}

  \pause
  \[
    \exists b \in B: \forall a \in A: f(a) \neq b
  \]
\end{frame}
%%%%%%%%%%%%%%%

%%%%%%%%%%%%%%%
\begin{frame}{}
  \begin{definition}[Bijective (one-to-one correspondence)]
    \centerline{Bijective: injective + surjective}
  \end{definition}
\end{frame}
