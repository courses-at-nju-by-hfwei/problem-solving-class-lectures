\begin{algorithm}[H]
  % \caption{Bellam-Ford Algorithm.}
  % \label{alg:bellman-ford}
  \begin{algorithmic}[1]
    \Procedure{Bellman-Ford}{$G, w, s$}
      \State \Call{Init-Single-Source}{$G, s$}

      \hStatex
      \State \red{$f \gets \textsc{False}$}
      \For{$i \gets 1 \;\text{\bf to}\; |V| - 1$}
	\For{$(u,v) \in E$}
	  \If{$v.d > u.d + w(u,v)$}
	    \State $v.d = u.d + w(u,v)$
	    \State \red{$f \gets \textsc{True}$}
	  \EndIf
	\EndFor

	\hStatex
	\If{\red{$f = \textsc{False}$}}
	  \State \Return
	\EndIf
      \EndFor
    \EndProcedure
  \end{algorithmic}
\end{algorithm}
