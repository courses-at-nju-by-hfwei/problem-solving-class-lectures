\begin{algorithm}[H]
  \begin{algorithmic}[1]
    \Procedure{mss}{$A, n$}
      \State $\mssat[0] \gets A[0]$

      \hStatex
      \ForAll{$i \gets 1 \ldots n-1$}
        \State $\mssat[i] \gets \max\set{\mssat[i-1] + A[i], 0}$
      \EndFor

      \hStatex
      \State \Return $\max\limits_{0 \le i \le n-1} \mssat[i]$
    \EndProcedure
  \end{algorithmic}
\end{algorithm}
