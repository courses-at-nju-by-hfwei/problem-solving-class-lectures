\begin{frame}
  \begin{definition}[Polynomial-time Reduction]
	\begin{center}
	  $L_1 \pr L_2$ if $\exists$ \red{poly.} time function $f$ such that
	\end{center}
	\[
	  \forall x: x \in L_1 \iff f(x) \in L_2.
	\]
  \end{definition}

  \pause
  \vspace{0.30cm}
  \[
	\forall L \in \np, \red{L \pr L'} \implies L' \text{ is } \nph
  \]

  \[
	\npc = \np \cap \nph
  \]
\end{frame}
