\begin{frame}{}

    \column{0.20\textwidth}
    \column{0.60\textwidth}
      \begin{algorithm}[H]
      \begin{algorithmic}[1]
	\Procedure{Perms}{$A[], l$}
	  \If{$l = A.size - 1$}
	    \State {\bf print} $A$
	  \Else
	    \For{$i \gets l \;{\bf to}\; A.size - 1$}
	      \State \Call{Swap}{$A[i], A[l]$}
	      \State \Call{Perms}{$A, l + 1$}
	      \onslide<2->{\State \red{\Call{Swap}{$A[i], A[l]$}}}
	    \EndFor
	  \EndIf
	\EndProcedure
      \end{algorithmic}
      \end{algorithm}
    \column{0.20\textwidth}
  \end{columns}
