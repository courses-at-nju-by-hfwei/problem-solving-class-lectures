	\[
	  L \in \conp
	\]
	\[
	  \iff
	\]
	\begin{center}
	  \red{$\exists$} \blue{poly.} time \emph{verifier} $V(x,c)$ such that
	\end{center}
	\pause 
	\[
	  \forall x \in \set{0,1}^{\ast}: \red{x \notin L} \iff \red{\exists} c \text{ with } |c| = O(|x|^{k}), V(x,c) = 1.
	\]
  \end{definition}

  \pause

	{\large \conp-problems has short \red{counterexamples} that are easy to verify.}
  \end{center}
