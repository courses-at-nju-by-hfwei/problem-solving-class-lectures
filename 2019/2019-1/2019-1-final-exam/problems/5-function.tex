\begin{frame}{Problem 5: Function}
  \begin{problem}
    对任意函数 $f$ 和集合 $A$, $B$。
    \begin{enumerate}[(1)]
      \item 请证明 $f(A) \setminus f(B) \subseteq f(A \setminus B)$。
	当 $f$ 满足什么条件时，``$=$'' 成立? 请给出理由。
      \item 请证明 $f^{-1}(A \setminus B) = f^{-1}(A) \setminus f^{-1}(B)$。
    \end{enumerate}
  \end{problem}
