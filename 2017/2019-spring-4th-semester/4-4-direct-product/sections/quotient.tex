\begin{frame}
  \begin{definition}[Internal Direct Product (Equivalent)]
	Let $G$ be a group with \red{normal} subgroups $H$ and $K$ satisfying
	\[
	  G = HK
	\]
	\[
	  H \cap K = \set{e}
	\]
	Then, $G$ is the internal direct product of $H$ and $K$.
  \end{definition}

  \only<2>{\fig{width = 0.35\textwidth}{figs/get-it-kenan}}

  \only<3->{
	\vspace{0.60cm}
	\begin{theorem}[The Second Isomorphism Theorem \only<4->{\red{(Diamond Theorem)}}]
	  \[
		H \le G, N \triangleleft G \red{\implies} H/(H \cap N) \cong HN/N.
	  \]
	\end{theorem}
  }
\end{frame}

  \begin{theorem}
	\begin{center}
	  If $G$ is the internal direct product of its normal subgroups $H$ and $K$, \\[5pt]
	  then $G/H \cong K, \quad G/K \cong H$.	
	\end{center}
  \end{theorem}
