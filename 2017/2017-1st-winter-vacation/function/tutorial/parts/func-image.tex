\begin{frame}{}
  \[
    f: X \to Y \quad A \subseteq X \quad B \subseteq Y
  \]

  \begin{definition}[Image]
    The \blue{image} of $A$ under $f$ is the set

    \[
      f(A) = \set{f(a) \mid a \in A}.
    \]
  \end{definition}

  \vspace{0.60cm}
  \begin{definition}[Inverse Image]
    The \blue{inverse image} of $B$ under $f$ is the set
    \[
      f^{-1}(B) = \set{x \in X \mid f(x) \in B}.
    \]
  \end{definition}
\end{frame}
