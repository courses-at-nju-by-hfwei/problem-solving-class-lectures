\begin{algorithm}[H]
  % \caption{LIS-Binary-Search.}
  % \label{alg:lis-binary-search}
  \begin{algorithmic}[1]
    \Procedure{\textsl{LIS}}{$n$}
      \State $E[l] \gets \emptyset, \quad \forall 1 \le i \le n$

      \hStatex
      \For{$i \gets 1 \;\uparrow\; n$}
      \State $\forall 1 \le l \le i : E[l] \gets$  \only<1>{\Comment{\it By extending each shorter subseq.}} \only<2->{\Comment{\teal{\it Extending only one subseq using binary search}}}
	\State \quad \text{\it the inc. subseq. of length $l$ \red{with the smallest ending number}}
      \EndFor

      \hStatex
      \State \Return $\max \set{l \mid E[l] \neq \emptyset}$
    \EndProcedure
  \end{algorithmic}
\end{algorithm}
