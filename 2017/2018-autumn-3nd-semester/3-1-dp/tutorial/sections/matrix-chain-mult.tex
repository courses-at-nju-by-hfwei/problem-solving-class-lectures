\begin{frame}{}
  \[
    m[i,j]: \min \text{ cost to \hl{yellow}{compute the matrix $A_{i} \cdots A_j$}}
  \]

  \[
    m[1,n]
  \]

  \pause
  \[
    \red{\fbox{\it Where is the {\teal{last}} parentheses?}}
  \]

  \pause
  \[
    m[i,j] = \min_{i \le k <j} \Big(m[i,k] + m[k+1,j] + p_{i-1}p_{k}p_{j}\Big)
  \]

  \pause
  \[
    m[i,i] = 0
  \]

  \pause
  \[
    O(n^3) = \Theta(n^2) \cdot O(n)
  \]
\end{frame}

  \]
