  \begin{definition}[Shortest Path]
    \[
      G = (V, E, w): \text{ weighted digraph}
    \]

    \[
      \delta(u,v) = \begin{cases}
	\min \big\{ w(p): u \leadsto^{p} v \big\} & \text{if } u \leadsto v \\
	\infty	& \text{o.w.}
      \end{cases}
    \]
  \end{definition}
