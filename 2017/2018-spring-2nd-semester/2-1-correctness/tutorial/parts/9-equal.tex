\begin{frame}[fragile]{}
  \begin{lstlisting}[style = CStyle]
    $X \gets S_1$
    $Y \gets S_2$

    $E \gets \top$
    while ($X \neq \Lambda \land E == \top$)
      if ($Y == \Lambda$)
        $E \gets \bot$
      else if $\texttt{eq}(\texttt{head}(X), \texttt{head}(Y))$
          $X \gets \texttt{tail}(X)$
          $Y \gets \texttt{tail}(Y)$
      else
        $E \gets \bot$

    if $Y \neq \Lambda$
      $E \gets \bot$

    return $E$
  \end{lstlisting}
\end{frame}
