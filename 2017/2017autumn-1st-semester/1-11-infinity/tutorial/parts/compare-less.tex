  \begin{definition}[Countable Revisited]
    $X$ is countable:
    \[
      \exists n \in \N: |X| = n \red{\;\lor\;} |X| = |\N|
    \]
  \end{definition}

  \begin{exampleblock}{UD Exercise $22.5$}
    $X$ is countable iff there exists a one-to-one function
    \[
      f: A \to \N.
    \]

    \[
      \blue{|X| \le |\N|.}
    \]
  \end{exampleblock}
