\begin{frame}{}
  \begin{definition}[Perfect Matching $\mathcal{M}$]
    \begin{center}
      Undirected weighted graph with $n = 2k$ vertices \\[15pt] \pause

      $\mathcal{M}:$ a set of $n/2$ edges such that
      \[
	\red{\boxed{\forall e, e' \in \mathcal{M}: e, e' \text{ have no common vertices}}}
      \]
    \end{center}

    \fig{width = 0.25\textwidth}{figs/perfect-matching-petersen}
  \end{definition}
\end{frame}

  \vspace{0.30cm}
  \begin{definition}[Minimum-weight Perfect Matching]
    \[
      w(\mathcal{M}) = \sum_{e \in \mathcal{M}} w(e)
    \]

    \vspace{0.20cm}
    \[
      \red{\min w(\mathcal{M})}
    \]
  \end{definition}
