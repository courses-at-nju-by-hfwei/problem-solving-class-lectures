\begin{frame}{}
  \begin{description}
    \item[Subproblem:] $m[i,j]:$ $\min$ cost to \hl{yellow}{compute the matrix $A_{i \dots j}$}
    \item[Goal:] $m[1,n]$
      \pause
      \vspace{0.30cm}
    \item[Make Choice:] \hl{yellow}{\red{Where} is the {\it \teal{last}} parentheses?}
    \item[Recurrence:] 
      \[
	m[i,j] = \min_{i \le k <j} \Big(m[i,k] + m[k+1,j] + p_{i-1}p_{k}p_{j}\Big)
      \]
      \pause
    \item[Init:]
      \[
	m[i,i] = 0
      \]
      \pause
    \item[Time:] 
      \[
	O(n^3) = \Theta(n^2) \cdot O(n)
      \]
  \end{description}
\end{frame}
