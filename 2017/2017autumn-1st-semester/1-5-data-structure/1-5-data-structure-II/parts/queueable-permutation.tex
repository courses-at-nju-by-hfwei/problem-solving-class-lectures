\begin{frame}
  \begin{theorem}[Queueable Permutations]
     A permutation $(a_1, \cdots, a_n)$ is queueable $\iff$ it is not the case that
     \[
       321\text{\it -Pattern}: \fbox{$\texttt{\it out} = \cdots a_i \cdots a_j \cdots a_k: i < j < k \land a_i > a_j > a_k$}
     \]
  \end{theorem}
