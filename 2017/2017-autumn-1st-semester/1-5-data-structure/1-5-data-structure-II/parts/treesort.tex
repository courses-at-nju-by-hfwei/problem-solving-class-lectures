\begin{frame}[fragile]{}
  \begin{exampleblock}{DH 2.16: Treesort}
    \begin{enumerate}[(i)]
      \item Construct an algorithm that transforms a given list of integers into a binary search tree.
    \end{enumerate}
  \end{exampleblock}

  \begin{columns}
    \column{0.48\textwidth}
      \pause
      \begin{lstlisting}[style = Cstyle]
|\purple{Node:}|
  int val = NIL,
  Node left = NULL,
  Node right = NULL
      \end{lstlisting}

      \pause
      \begin{lstlisting}[style = Cstyle]
|\purple{buildBST}|(int eles[]):
  Node root(eles[0])

  foreach e $\in$ eles[1..]:
    insert(root, e)
      \end{lstlisting}
    \column{0.52\textwidth}
      \pause
      \begin{lstlisting}[style = Cstyle]
|\purple{insert}|(Node T, int e):  |\pause|
  if (e < T.val)
    if (T.left == NULL)
      T.left = new Node(e)
    else
      insert(T.left, e)  |\pause|
  else  // e >= T.val
    if (T.right == NULL)
      T.right = new Node(e)
    else
      insert(T.right, e)
      \end{lstlisting}
  \end{columns}
\end{frame}
