	  \item \textcolor<2->{red}{$A \cap B \setminus (A \cap B \cap C)$}
	  \item $A \cap B \cap C^{c}$
	  \item $(A \cap B) \setminus C$
	  \item $(A \setminus C) \cap (B \setminus C)$
	\end{enumerate}
      \column{0.50\textwidth}
	\uncover<3->{
