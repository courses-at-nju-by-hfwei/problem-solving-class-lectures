\begin{algorithm}[H]
  \begin{algorithmic}[1]
    \Procedure{mss}{$A, n$}
      \State \blue{$\mssat[0] \gets A[0]$}

      \hStatex
      \For{$i \gets 1 \ldots n-1$}
        \State \red{$\mssat[i] \gets \max\set{\mssat[i-1] + A[i], A[i]}$}
      \EndFor

      \hStatex
      \State \Return \blue{$\max\limits_{0 \le i \le n-1} \mssat[i]$}
    \EndProcedure
  \end{algorithmic}
\end{algorithm}
